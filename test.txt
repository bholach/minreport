import java.util.HashSet;
import java.util.Iterator;

public class Counries {

	HashSet<String> country = new HashSet<String>();
	
	public HashSet<String> saveCountryName(String CountryName) { 
		country.add(CountryName);
		
		return country;
	}
	public String getCountry(String CountryName,HashSet<String> country) {
		
		boolean flag=false;
		String cname="";
		Iterator<String> itr=country.iterator();  
		  while(itr.hasNext()){  
			  cname=itr.next();
		   if(cname.equalsIgnoreCase(CountryName))
		     { 
			     flag=true;
		         break;
		     }
		   }
		  if(flag) {
			  return cname;
		  }
		  else
		     return null;
		
	}
	public static void main(String args[]){  
		  //Creating HashSet and adding elements  
		 HashSet<String> set = new HashSet<String>();  
		 Counries country = new Counries(); 
		 set = country.saveCountryName("india");
		 set = country.saveCountryName("us");
		 set = country.saveCountryName("america");
		 System.out.println(country.getCountry("us",set));
		  
		 }  

}
